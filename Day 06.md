## [P51. The missing one](https://codeforces.com/group/yg7WhsFsAp/contest/355504/problem/P51)
```
#include<bits/stdc++.h>
using namespace std;
int main(){
    cin.tie(0)->sync_with_stdio(0);
    int n,x;
    cin>>n;
    x= n*(n+1)/2;
    --n;
    while(n--){
        int y;  cin>>y;
        x-=y;
    }
    cout<<x<<endl;
    return 0;
}
```
